-- Install language servers for languages you use
:LspInstall bashls cssls dockerls gopls jsonls tsserver sumneko_lua pylsp rust_analyzer svelte taplo tailwindcss vimls volar yamlls

:PylspInstall pyls-flake8 pyls-mypy pyls-isort

-- Install treesitter parsers for languages you use
:TSInstall bash css dockerfile go gomod html javascript json lua markdown python regex rust svelte toml typescript vim vue yaml
